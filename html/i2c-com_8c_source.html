377void i2c_set_speed(I2C_TypeDef_t* i2c, enum i2c_speeds speed, uint32_t clock_megahz)
378{
379 i2c_set_clock_frequency(i2c, clock_megahz);
380 switch(speed)
381 {
382 case i2c_speed_fm_400k:
383 i2c_set_fast_mode(i2c);
384 i2c_set_ccr(i2c, clock_megahz * 5 / 6);
385 i2c_set_trise(i2c, clock_megahz + 1);
386 break;
387
388 case i2c_speed_sm_100k:
389 default:/* fall back to standard mode */
390 i2c_set_standard_mode(i2c);
391 i2c_set_ccr(i2c, clock_megahz * 5); /* x Mhz / (100kHz * 2) */
392 i2c_set_trise(i2c, clock_megahz + 1); /* Sm mode, (100kHz) freqMhz + 1 */
393 break;
394 }
395}

538uint8_t i2c_write7_v1(I2C_TypeDef_t* i2c, int addr, const uint8_t *data, size_t n)
539{
540 I2Cx_Timeout=I2C_MAX_BUSY_TIME;
541 // wait while busy
542 while(i2c->SR2.BUSY) //((I2C_SR2(i2c) & I2C_SR2_BUSY))
543 if(I2Cx_Timeout==0)
544 return 1;
545
546 // send start
547 i2c_send_start(i2c);
548 I2Cx_Timeout=I2C_MAX_WAIT_TIME;
549 /* Wait for the end of the start condition, master mode selected, and BUSY bit set */
550 while(!i2c->SR1.SB && i2c->SR2.MSL && i2c->SR2.BUSY) //while( !( (I2C_SR1(i2c) & I2C_SR1_SB) && (I2C_SR2(i2c) & I2C_SR2_MSL) && (I2C_SR2(i2c) & I2C_SR2_BUSY) ));
551 if( I2Cx_Timeout==0)
552 return 0;
553
554 // send the slave address
555 i2c_send_7bit_address(i2c, addr, I2C_WRITE);
556
557 /* Waiting for address is transferred. */
558 //while (!(I2C_SR1(i2c) & I2C_SR1_ADDR));
559 I2Cx_Timeout=I2C_MAX_WAIT_TIME;
560 while(!i2c->SR1.ADDR)
561 if( I2Cx_Timeout==0)
562 return 1;
563
564 /* Clearing ADDR condition sequence. */
565 (void)i2c->SR1.val;
566 (void)i2c->SR2.val; //(void)I2C_SR2(i2c);
567 // send the array of data
568 for (size_t i = 0; i < n; i++)
569 {
570 i2c_send_data(i2c, data[i]);
571 I2Cx_Timeout=I2C_MAX_WAIT_TIME;
572 while(!i2c->SR1.BTF)// while (!(I2C_SR1(i2c) & (I2C_SR1_BTF)));
573 if( I2Cx_Timeout==0)
574 return 1;
575 }
576 // send stop condition
577 i2c_send_stop(i2c);
578 return 1;
579}

581uint8_t i2c_read7_v1(I2C_TypeDef_t* i2c, int addr, uint8_t *res, size_t n)
582{
583 I2Cx_Timeout=I2C_MAX_BUSY_TIME;
584
585 i2c_send_start(i2c);
586 i2c_enable_ack(i2c);
587
588 /* Wait for the end of the start condition, master mode selected, and BUSY bit set */
589 //while ( !( (I2C_SR1(i2c) & I2C_SR1_SB) && (I2C_SR2(i2c) & I2C_SR2_MSL) && (I2C_SR2(i2c) & I2C_SR2_BUSY) ));
590
591 while(!i2c->SR1.SB && i2c->SR2.MSL && i2c->SR2.BUSY)
592 {
593 if(I2Cx_Timeout==0)
594 return 1;
595 }
596 i2c_send_7bit_address(i2c, addr, I2C_READ);
597 /* Waiting for address is transferred. */
598 //while (!(I2C_SR1(i2c) & I2C_SR1_ADDR));
599 I2Cx_Timeout=I2C_MAX_WAIT_TIME;
600 while(!i2c->SR1.ADDR)
601 {
602 if(I2Cx_Timeout==0)
603 // set error and exit
604 return 1;
605 }
606 /* Clearing ADDR condition sequence. */
607 (void)i2c->SR1.val;
608 (void)i2c->SR2.val; // (void)I2C_SR2(i2c);
609
610 for (size_t i = 0; i < n; ++i)
611 {
612 if (i == n - 1)
613 {
614 i2c_disable_ack(i2c);
615 }
616 //while (!(I2C_SR1(i2c) & I2C_SR1_RXNE));
617 I2Cx_Timeout=I2C_MAX_WAIT_TIME;
618 while(!i2c->SR1.RXNE)
619 {
620 if(I2Cx_Timeout==0)
621 // set error and exit
622 return 1;
623
624 }
625 res[i] = i2c_get_data(i2c);
626 }
627 // send stop condition
628 i2c_send_stop(i2c);
629
630 return 0;
631}
