216uint8_t lcd_read_byte(void)
217{
218 uint8_t val=0;
219
220 //set LCD port as an input port
221 lcd_port_pinmode_in();
222
223 // select read operation on GLCD
224 //rw.set();
225
226 //enable lcd to read
227 lcd_en_set(); //en.set();
228 // wait dummy cycles before read
229
230 asm("nop");
231 asm("nop");
232 asm("nop");
233 asm("nop");
234
235 _delay_us(1);
236
237 // Read a value to GLCD Port
238 val=lcd_port_read_byte();
239
240 // end glcd to read operation
241 lcd_en_clear(); //en.clear();

481{
482 uint8_t i=0;
483 uint8_t Dig[3]={0};

486 Dig[i]=val%10;
487 val/=10;

493 lcd_write_data(bcd_tbl[Dig[1]]);
494 lcd_write_data(bcd_tbl[Dig[0]]);
495}

617size_t lcd_print_as_number(unsigned long long n, uint8_t base)
618{
619 char buf[8 * sizeof(long long) + 1]; // Assumes 8-bit chars plus zero byte.
620 char *str = &buf[sizeof(buf) - 1];
621
622 *str = '\0';
623
624 // prevent crash if called with base == 1
625 if (base < 2) base = 10;
626
627 do {
628 char c = n % base;
629 n /= base;
630
631 *--str = c < 10 ? c + '0' : c + 'A' - 10;
632 } while(n);
633
634 return lcd_write_cstr(str);
635}
636//--------------------------------------------------------------------------------------------
637size_t lcd_print_as_Float(double number, uint8_t digits)
638{
639 size_t n = 0;
640 uint8_t i=0;
641
642 if (isnan(number)) return lcd_print("nan");
643 if (isinf(number)) return lcd_print("inf");
644 if (number > 4294967040.0) return lcd_print ("ovf"); // constant determined empirically
645 if (number <-4294967040.0) return lcd_print ("ovf"); // constant determined empirically
646
647 // Handle negative numbers
648 if (number < 0.0)
649 {
650 n += lcd_print_ch('-');
651 number = -number;
652 }
653
654 // Round correctly so that print(1.999, 2) prints as "2.00"
655 double rounding = 0.5;
656 for (i=0; i<digits; ++i)
657 rounding /= 10.0;
658
659 number += rounding;
660
661 // Extract the integer part of the number and print it
662 unsigned long int_part = (unsigned long)number;
663 double remainder = number - (double)int_part;
664 n += lcd_print_int64(int_part,10);
665
666 // Print the decimal point, but only if there are digits beyond
667 if (digits > 0) {
668 n += lcd_print_ch('.');
669 }
670
671 // Extract digits from the remainder one at a time
672 while (digits-- > 0)
673 {
674 remainder *= 10.0;
675 unsigned int toPrint = (unsigned int)(remainder);
676 n += lcd_print_uint16((uint16_t)toPrint,10);
677 remainder -= toPrint;
678 }
679
680 return n;
681}
