94real_t avg_real(real_t a,real_t b)
95{
96 real_t c;
97 int16_t d1=0,d2=0,d3=0;
98 c.val=0;
99 d1=a.integer*10+a.decimal;
100 d2=b.integer*10+b.decimal;
101 d1=(a.sig==1)?-d1:d1;
102 d2=(b.sig==1)?-d2:d2;
103 d3=d1+d2;
104 c.sig=0;
105 if(d3<0)
106 {
107 c.sig=1;
108 d3*=-1;
109 }
110 c.integer=d3/20;
111 c.decimal=(d3-c.integer*20)/2;
112 return c;
113}

365real_t float_to_real(float number,uint8_t digits)
366{
367 real_t r;
368 r.val=0;
369 if(number<0)
370 {
371 r.sig=1;
372 number=-number;
373 }
374
375 // Round correctly so that print(1.999, 2) prints as "2.00"
376 double rounding = 0.5;
377 for (uint8_t i=0; i<digits; ++i)
378 rounding /= 10.0;
379
380 number += rounding;//number += 0.05f;
381
382 // Extract the integer part of the number and print it
383 unsigned long int_part = (unsigned long)number;
384 float reminder = number - (double)int_part;
385
386 // Extract digits from the remainder one at a time
387 while (digits-- > 0)
388 reminder *=10.0f;
389
390 r.integer=int_part;
391 r.decimal=(uint8_t) reminder;
392
393 return r;
394}

404float real_to_float(real_t r,uint8_t digits)
405{
406 float number=0,reminder=0;
407 number=r.integer;
408 reminder = (float)r.decimal;
409
410 for (uint8_t i=0; i<digits; ++i)
411 reminder /= 10.0;
412
413 number+=reminder;
414
415 if(r.sig==1)
416 number=-number;
417 return number;
418}
